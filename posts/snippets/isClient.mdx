### isServer
```ts
declare const global: unknown;

export function isServer() {
    return typeof window === 'undefined' && typeof global !== 'undefined';
}
```

### isClient
```ts
export function isClient() {
    return !isServer()
}
```

----

```ts
if(client()) {
    // 브라우저 환경일때만 실행되는게 보장됨
```
